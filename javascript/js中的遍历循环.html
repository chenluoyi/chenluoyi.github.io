let arr = [1,2,3,4,5];

for(let i = 0; i < arr.length; i++){

    # 二、forEach循环

    • 没有返回值
    • 循环数组中的每一个元素并采取操作
    • 实际效率并没有for循环高
    • 参数:第一个是循环的每一项item，第二个参数是每一项对应的下标index
    //假设存在一个数组arr
let arr = [1,2,3,4,5];

//forEach循环
arr.forEach((item,index) => {
    //do someing……
    console.log(arr[i]);
});

//console.log
1
2
3
4
5

    # 五、filter过滤器

    • 遍历数组并返回一个新的数组，不会影响原数组，遍历数组每一项，回调函数返回true
    • 回调函数有两个参数(item,index)，item是每一项，index是下标
    //假设存在一个数组arr
let arr = [1,2,3,4,5];
//声明一个新的空数组
let newArr = [];
//给数组添加一个私有属性
arr.a = "key"

//filter过滤器
//这里使用的是es6箭头函数
newArr = arr.filter(item=>item>3);
//这里使用的是es5语法,和上面的操作相等
newArr = arr.filter(function(item){
    return item > 3;
})
console.log(newArr)

//console.log
[4,5]

    # 六、map映射

    • 去遍历数组的每一项，但是不会操作改变原数组，同filter一样会返回一个新的数组。回调函数返回的是什么，对应的新数组的那一项就会是什么
    //假设存在一个数组arr
let arr = [1,2,3,4,5];
//声明一个新的空数组
let newArr = [];
//给数组添加一个私有属性
arr.a = "key"

//map映射
//这里使用的是es6语法
newArr = arr.map(item=>`<div>${item}</div>`);
//这里是一般语法
newArr = arr.map(function(item){
    return "<div>" + item + "</div>";
})
console.log(newArr)

//console.log
[
    "<div>1</div>",
    "<div>2</div>",
    "<div>3</div>",
    "<div>4</div>",
    "<div>5</div>"
]

    # 七、include和find(ES6)

    • include有一个参数，它会遍历数组的每一项与参数对比，如果存在该参数，则返回true
    • find的参数是一个回调函数，函数的参数是每一项以及对应下标，find会返回匹配的那一项，一旦匹配就不会继续往下匹配,返回一个新的数组
    //假设存在一个数组arr
let arr = [1,2,3,4,5,55];

//include
console.log(arr.include(5));

//include console
true

//find
let res = arr.find(function(item,index){
    //如果这一项包含了5则返回true
    return item.toString().indexOf(5) > -1;
})
console.log(res);

//find console
5

    # 八、some和every

    • some遍历数组，找到true，即返回true，否则返回false
    • every则是找到false返回false，否则返回true
    //假设存在一个数组arr
let arr = [1,2,3,4,5,55];

//some
let someResult = arr.some(item=>item > 4);
console.log(someResult);

//some console
true

//every
let evevyResult = arr.every(item=>item > 0);
console.log(everyResult);

//every console
true
